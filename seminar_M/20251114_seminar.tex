\section*{雑談}
\begin{align}
  \int_{1}^{2}\left(x^2 + 3x\right)dx + \int_{1}^{2}\left(x^2 - 3x\right)dx &= \int_{1}^{2}\left\{\left(x^2 + 3x\right) + \left(x^2 - 3x\right)\right\}dx \\
  &= \int_{1}^{2}2x^2dx \\
  &= 2\left[\frac{x^3}{3}\right]^2_1 = \frac{2\left(2^3 - 1^3\right)}{3} = \frac{14}{3}
\end{align}
